Backend Google Apps Script

Script único para Drive + Sheets · Configure uma vez · Funciona para sempre

O que este script faz Cria a estrutura de pastas no Drive ("Plataforma de Estudos MEA" → Apostilas / Listas / Questionários / Análises) · Registra cada material no Sheets (matéria, banca, concurso, órgão, data, link) · Cria Google Forms quando solicitado · Gera análise de desempenho dos questionários.

Como Configurar
Cole este código no Google Apps Script (Code.gs)
Passo 1 — Criar a Planilha Acesse sheets.google.com → Criar nova planilha → Copie o ID da URL (entre /d/ e /edit) → Cole no campo acima.
Passo 2 — Abrir o Apps Script Na planilha: Extensões → Apps Script → cole o código gerado no Code.gs → Salve.
Passo 3 — Executar setup inicial No Apps Script, execute a função setupMEA() → Autorize as permissões → As abas e pastas serão criadas automaticamente.
Passo 4 — Usar desta plataforma Volte aqui → sempre que gerar um prompt e clicar "Registrar no Sheets", o material será catalogado. Use a função registerMaterial(dados) via URL fetch ou manualmente.
Atenção Esta plataforma HTML é local (roda no navegador). O registro no Sheets é feito copiando o script e executando manualmente no Apps Script, ou via Google Apps Script Web App (deploy como webapp para integração automática).

Script GAS Executor (Code.gs) — instale uma vez
// ════════════════════════════════════════════════════════ // MEA GAS EXECUTOR · Web App · AKE UFT-1.0 // Cole este código no Apps Script do seu Google Sheets // Depois: Implantar → Nova implantação → Web App // Executar como: "Eu" | Quem acessa: "Qualquer pessoa" // Copie a URL gerada e cole na plataforma HTML // ════════════════════════════════════════════════════════ var MEA_EXECUTOR_CONFIG = { pastaRaiz: "Plataforma de Estudos MEA", sheetId: PropertiesService.getScriptProperties().getProperty("MEA_SHEET_ID") || "", autor: PropertiesService.getScriptProperties().getProperty("MEA_AUTOR") || "Candidato MEA", token: PropertiesService.getScriptProperties().getProperty("MEA_TOKEN") || "mea2025" }; // ── WEB APP ENTRY POINT ────────────────────────────────── // CORS: GAS retorna Access-Control-Allow-Origin:* automaticamente // quando o output é ContentService com MimeType JSON. // O fetch da plataforma envia Content-Type: text/plain para // evitar preflight OPTIONS (GAS não suporta OPTIONS). function doPost(e) { try { var raw = e.postData ? e.postData.contents : '{}'; var payload = JSON.parse(raw); // Verificação de token if (payload.token !== MEA_EXECUTOR_CONFIG.token) { return jsonResponse({ success: false, error: 'Token inválido. Configure o token correto na plataforma.' }); } var resultado = executarScriptMEA(payload.script, payload.dados || {}); return jsonResponse(resultado); } catch (err) { return jsonResponse({ success: false, error: err.toString() }); } } // doGet: usado pelo botão "Testar conexão" da plataforma function doGet(e) { return jsonResponse({ status: 'MEA Executor online', versao: 'AKE UFT-1.0', timestamp: new Date().toISOString() }); } // Helper: ContentService JSON com CORS function jsonResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); // O GAS adiciona Access-Control-Allow-Origin: * automaticamente // para respostas JSON de Web Apps com acesso "Qualquer pessoa" } // ── EXECUTOR PRINCIPAL ─────────────────────────────────── function executarScriptMEA(scriptCode, dadosExtras) { try { // O script da IA deve terminar com: resultado = { success: true, docUrl: "...", ... } // Injetamos helpers disponíveis no contexto e executamos via eval var contexto = { SpreadsheetApp: SpreadsheetApp, DocumentApp: DocumentApp, DriveApp: DriveApp, FormApp: FormApp, Utilities: Utilities, Session: Session, Logger: Logger, dadosExtras: dadosExtras, garantirPasta: garantirPastaHelper, moverParaPasta: moverParaPastaHelper }; // Preparar o script: injetar helpers e capturar resultado var scriptWrapper = "var SpreadsheetApp = contexto.SpreadsheetApp;" + "var DocumentApp = contexto.DocumentApp;" + "var DriveApp = contexto.DriveApp;" + "var FormApp = contexto.FormApp;" + "var Utilities = contexto.Utilities;" + "var Session = contexto.Session;" + "var Logger = contexto.Logger;" + "var garantirPasta = contexto.garantirPasta;" + "var moverParaPasta = contexto.moverParaPasta;" + scriptCode + "\n; return typeof resultado !== 'undefined' ? resultado : { success: false, error: 'Script não retornou resultado' };"; var fn = new Function("contexto", scriptWrapper); var resultado = fn(contexto); if (resultado && resultado.success && resultado.docUrl) { registrarExecucao(resultado, dadosExtras); } return resultado || { success: false, error: "Nenhum resultado retornado" }; } catch (err) { Logger.log("executarScriptMEA erro: " + err.toString()); return { success: false, error: err.toString() }; } } // ── REGISTRAR NO SHEETS ────────────────────────────────── function registrarExecucao(resultado, dados) { try { var ss = MEA_EXECUTOR_CONFIG.sheetId ? SpreadsheetApp.openById(MEA_EXECUTOR_CONFIG.sheetId) : SpreadsheetApp.getActiveSpreadsheet(); var tipo = dados.tipo || "Geral"; var abaMap = { "Apostila": "Apostilas", "Guia": "Apostilas", "Lista": "Listas", "Lista de Exercícios": "Listas", "Questionário": "Questionários", "Análise de Banca": "Análises", "Análise": "Análises" }; var abaNome = abaMap[tipo] || "Listas"; var aba = ss.getSheetByName(abaNome); if (!aba) { aba = ss.insertSheet(abaNome); var cabec = ["ID","Tipo","Matéria","Banca","Cargo","Assunto","Versão","Data","Link Drive","Status"]; aba.getRange(1,1,1,cabec.length).setValues([cabec]) .setBackground("#162B20").setFontColor("#C8A838").setFontWeight("bold"); aba.setFrozenRows(1); } var id = tipo.charAt(0) + new Date().getTime(); aba.appendRow([ id, dados.tipo || "Geral", dados.materia || "", dados.banca || "", dados.cargo || "", dados.assunto || "", dados.versao || "v1", new Date().toLocaleDateString("pt-BR"), resultado.docUrl || "", "OK" ]); atualizarDashboardExecutor(ss); Logger.log("Registrado: " + id + " | " + resultado.docUrl); } catch (e) { Logger.log("registrarExecucao erro: " + e.toString()); } } // ── DASHBOARD ──────────────────────────────────────────── function atualizarDashboardExecutor(ss) { try { var dash = ss.getSheetByName("Dashboard"); if (!dash) { dash = ss.insertSheet("Dashboard"); } var abas = ["Apostilas","Listas","Questionários","Análises"]; var rows = [["📊 DASHBOARD MEA","Atualizado: " + new Date().toLocaleString("pt-BR")]]; abas.forEach(function(a){ var sh = ss.getSheetByName(a); rows.push([a, sh ? Math.max(0, sh.getLastRow()-1) : 0]); }); dash.clearContents(); dash.getRange(1,1,rows.length,2).setValues(rows); dash.getRange(1,1,1,2).setBackground("#162B20").setFontColor("#C8A838").setFontWeight("bold"); } catch(e) { Logger.log("Dashboard erro: "+e.toString()); } } // ── HELPERS DE PASTA ───────────────────────────────────── function garantirPastaHelper(caminho) { var partes = (caminho||"").split("/").map(function(p){ return p.trim(); }).filter(Boolean); var pasta = DriveApp.getRootFolder(); partes.forEach(function(p){ var it = pasta.getFoldersByName(p); pasta = it.hasNext() ? it.next() : pasta.createFolder(p); }); return pasta; } function moverParaPastaHelper(fileId, caminhoOuSubpasta) { try { var caminho = caminhoOuSubpasta.indexOf("/") !== -1 ? caminhoOuSubpasta : MEA_EXECUTOR_CONFIG.pastaRaiz + "/" + caminhoOuSubpasta; var pasta = garantirPastaHelper(caminho); var file = DriveApp.getFileById(fileId); pasta.addFile(file); try { var pais = file.getParents(); while (pais.hasNext()) { var pai = pais.next(); if (pai.getId() !== pasta.getId()) pai.removeFile(file); } } catch(e) {} Logger.log("Arquivo movido para: " + caminho); return pasta.getUrl(); } catch(e) { Logger.log("moverParaPasta erro: " + e.toString()); return ""; } } // ── SETUP INICIAL ──────────────────────────────────────── function setupExecutor() { var ui = SpreadsheetApp.getUi(); var token = ui.prompt("🔐 Defina um token de segurança:", "mea2025", ui.ButtonSet.OK_CANCEL); if (token.getSelectedButton() !== ui.Button.OK) return; var sheetId = SpreadsheetApp.getActiveSpreadsheet().getId(); var autor = ui.prompt("👤 Seu nome:", "Candidato MEA", ui.ButtonSet.OK_CANCEL); PropertiesService.getScriptProperties().setProperties({ MEA_TOKEN: token.getResponseText() || "mea2025", MEA_SHEET_ID: sheetId, MEA_AUTOR: autor.getSelectedButton() === ui.Button.OK ? autor.getResponseText() : "Candidato MEA" }); // Criar pastas e abas var pastas = ["Apostilas","Listas","Questionários","Análises"]; pastas.forEach(function(p){ garantirPastaHelper("Plataforma de Estudos MEA/" + p); }); var ss = SpreadsheetApp.getActiveSpreadsheet(); var abas = { "Apostilas": ["ID","Tipo","Matéria","Banca","Cargo","Assunto","Versão","Data","Link Drive","Status"], "Listas": ["ID","Tipo","Matéria","Banca","Cargo","Assunto","Versão","Data","Link Drive","Status"], "Questionários": ["ID","Tipo","Matéria","Banca","Qtd","Data","PDF","Forms","Status"], "Análises": ["ID","Banca","Cargo","Matéria","Nível","Data","Link","Status"], "Dashboard": ["Aba","Total"] }; Object.keys(abas).forEach(function(nome){ var sh = ss.getSheetByName(nome) || ss.insertSheet(nome); if (sh.getLastRow() === 0) { sh.getRange(1,1,1,abas[nome].length).setValues([abas[nome]]) .setBackground("#162B20").setFontColor("#C8A838").setFontWeight("bold"); sh.setFrozenRows(1); } }); ui.alert("✅ MEA Executor configurado!\n\nPróximo passo:\nImplantar → Nova implantação → Web App\n Executar como: Eu\n Quem acessa: Qualquer pessoa\n\nCopie a URL e cole na plataforma HTML."); } // ── MENU ───────────────────────────────────────────────── function onOpen() { try { SpreadsheetApp.getUi() .createMenu("🎯 MEA Executor") .addItem("⚙ Configurar Executor", "setupExecutor") .addItem("📊 Atualizar Dashboard", function(){ atualizarDashboardExecutor(SpreadsheetApp.getActiveSpreadsheet()); }) .addToUi(); } catch(e) {} }
